\title{Equilibrium lesser Green's function of the YSR Josephson contact:\\

\noindent
This note derives the equilibrium ($V=0$, $T=0$) lesser Green's function used in
\texttt{currentPhi\_eq\_Tfull} (the current--phase relation $I(\phi)$ of a
current--biased/equilibrium Josephson contact with a classical--spin/YSR impurity,
in the $4\times4$ Nambu$\otimes$spin basis). It shows that the compact ``FDT'' form
\[
  G^{<} \;=\; ff\cdot\bigl(\GR{}\dg - \GR\bigr),\qquad ff \equiv \theta(-\omega),
\]
is \emph{algebraically identical} to the explicit bath construction
$G^{<}=\GR\,\Sless\,\GA$ with a specific, clean $\Sless$ --- and in particular that
the impurity potential, the tunnelling, and any flat ``$2i\Gamma$'' term must
\emph{not} appear in $\Sless$.

\section{Model and notation}

At real frequency $\omega$ define $z=\omega+i\Gamma$ ($\Gamma$ = Dynes broadening).
The clean BCS surface Green's function of each lead is
\begin{equation}
  \gz(z)=\frac{1}{\zeta\sqrt{\Delta^{2}-z^{2}}}
  \begin{pmatrix}
    -z & 0 & 0 & \Delta\\
    0 & -z & -\Delta & 0\\
    0 & -\Delta & -z & 0\\
    \Delta & 0 & 0 & -z
  \end{pmatrix},
  \qquad
  \gz^{A}\equiv \gz\dg=\gz(z^{*}),\ \ z^{*}=\omega-i\Gamma .
\end{equation}
The on--site impurity is Hermitian,
\begin{equation}
  \Vimp=\begin{pmatrix}\mathbf{J}\!\cdot\!\boldsymbol\sigma & 0\\[2pt]
        0 & -(\mathbf{J}\!\cdot\!\boldsymbol\sigma)^{*}\end{pmatrix}
        + K\,\tz\!\otimes\!\sigma_{0},
  \qquad \Vimp\dg=\Vimp .
\end{equation}
The impurity--dressed retarded surface GF of lead $\alpha=L,R$ is
\begin{equation}
  g_{r\alpha}=\bigl(\mathbb{I}-\gz\Vimp^{\alpha}\bigr)^{-1}\gz
  \quad\Longrightarrow\quad
  g_{r\alpha}^{-1}=\gz^{-1}-\Vimp^{\alpha}.
  \label{eq:grinv}
\end{equation}
The two leads are stacked block--diagonally, $g_{\mathrm{rj}}=\mathrm{diag}(g_{rL},g_{rR})$,
and coupled by the (transparency $\mathcal T$) tunnelling vertex
\begin{equation}
  \Sigma_{\mathrm{rj}}=\begin{pmatrix}0 & \Sigma_{lr}\\ \Sigma_{rl} & 0\end{pmatrix},
  \qquad \Sigma_{lr}=\Sigma_{rl}\dg
  \ \Longrightarrow\ \Sigma_{\mathrm{rj}}\dg=\Sigma_{\mathrm{rj}} .
  \label{eq:tunnelherm}
\end{equation}
The full retarded GF is the Dyson solution
\begin{equation}
  \GR\equiv G_{\mathrm{rjwar}}
  =\bigl(\mathbb{I}-g_{\mathrm{rj}}\Sigma_{\mathrm{rj}}\bigr)^{-1}g_{\mathrm{rj}}
  =\bigl(g_{\mathrm{rj}}^{-1}-\Sigma_{\mathrm{rj}}\bigr)^{-1},
  \qquad
  \GR{}^{-1}=g_{\mathrm{rj}}^{-1}-\Sigma_{\mathrm{rj}} .
  \label{eq:Ginv}
\end{equation}

\section{The equilibrium requirement}

In equilibrium the Keldysh (Dyson) form and the fluctuation--dissipation form of the
lesser GF must coincide,
\begin{equation}
  G^{<}=\GR\,\Sless\,\GA
  \qquad\text{and}\qquad
  G^{<}=f\,(\GA-\GR),
  \label{eq:two}
\end{equation}
with $f(\omega)=\theta(-\omega)=ff$ at $T=0$. The two agree \emph{iff} the lesser
self--energy is the FDT partner of the retarded one,
\begin{equation}
  \Sless=f\,(\Sigma^{A}-\Sigma^{R}),
  \qquad
  \Sigma^{A}-\Sigma^{R}=\GR{}^{-1}-\GA{}^{-1},
  \label{eq:fdtSE}
\end{equation}
the last equality because the bare (Hermitian) inverse propagator cancels in
$\Sigma^{A}-\Sigma^{R}$. Equation~\eqref{eq:fdtSE} is the single condition a correct
$\Sless$ must satisfy.

\section{What $ff\,(\GA-\GR)$ evaluates to}

Start from the algebraic identity, valid for any invertible $G$,
\begin{equation}
  \GA-\GR=\GR\bigl(\GR{}^{-1}-\GA{}^{-1}\bigr)\GA ,
  \label{eq:keldysh}
\end{equation}
which one checks by expanding
$\GR\GR{}^{-1}\GA-\GR\GA{}^{-1}\GA=\GA-\GR$. Multiplying by $ff$,
\begin{equation}
  ff\,(\GA-\GR)=\GR\,\underbrace{\bigl[\,ff\,(\GR{}^{-1}-\GA{}^{-1})\,\bigr]}_{\displaystyle \equiv\,\Sless_{A}}\,\GA .
  \label{eq:tobath}
\end{equation}
So the FDT form \emph{is} the bath form, with effective self--energy
$\Sless_{A}=ff\,(\GR{}^{-1}-\GA{}^{-1})$. Now evaluate it using \eqref{eq:Ginv}.
Two cancellations occur.

\medskip
\noindent\textbf{(i) The tunnelling drops out.} Since $\Sigma_{\mathrm{rj}}$ is
Hermitian~\eqref{eq:tunnelherm},
\begin{equation}
  \GR{}^{-1}-\GA{}^{-1}
  =\bigl(g_{\mathrm{rj}}^{-1}-\Sigma_{\mathrm{rj}}\bigr)
  -\bigl(g_{\mathrm{rj}}^{A\,-1}-\Sigma_{\mathrm{rj}}\bigr)
  =g_{\mathrm{rj}}^{-1}-g_{\mathrm{rj}}^{A\,-1}.
\end{equation}

\noindent\textbf{(ii) The impurity drops out.} Per lead, using \eqref{eq:grinv} and
$\Vimp\dg=\Vimp$,
\begin{equation}
  g_{r\alpha}^{-1}-g_{a\alpha}^{-1}
  =\bigl(\gz^{-1}-\Vimp^{\alpha}\bigr)-\bigl(\gz^{A\,-1}-\Vimp^{\alpha}\bigr)
  =\gz^{-1}-\gz^{A\,-1}.
\end{equation}
Both leads give the \emph{same} block, so
\begin{equation}
  \Sless_{A}=ff\,
  \begin{pmatrix}\gz^{-1}-\gz^{A\,-1} & 0\\[2pt] 0 & \gz^{-1}-\gz^{A\,-1}\end{pmatrix}.
  \label{eq:SAmid}
\end{equation}

\section{The surface identity}

The BCS surface GF obeys the recursion $\gz^{-1}=Q(z)-\zeta^{2}\,\tz\,\gz\,\tz$ with
$Q(z)=Q(z)\dg$ Hermitian (equivalently $\gz^{-1}+\zeta^2\tz\gz\tz$ is Hermitian;
verified numerically to machine precision, $\lesssim10^{-15}$). Subtracting the
advanced version, the Hermitian $Q$ cancels and
\begin{equation}
  \boxed{\;\gz^{-1}-\gz^{A\,-1}
  =\zeta^{2}\,\tz\,(\gz^{A}-\gz)\,\tz\;}
  \label{eq:surfid}
\end{equation}
i.e.\ the \emph{entire} anti--Hermitian part of $\gz^{-1}$ is the reservoir term;
there is no separate flat $-i\Gamma$ piece. Writing the clean spectral weight
$g\ell_{0}\equiv-(\gz-\gz^{A})=\gz^{A}-\gz$ and inserting \eqref{eq:surfid} into
\eqref{eq:SAmid},
\begin{equation}
  \boxed{\;\Sless_{A}=ff\;\zeta^{2}\,\tz\,g\ell_{0}\,\tz\quad\text{per lead},
  \qquad g\ell_{0}=-(\gz-\gz^{A}),\ \ \gz=\gz(\omega+i\Gamma).\;}
  \label{eq:result}
\end{equation}

\section{Comparison with the explicit \texttt{Siglj}}

Equations \eqref{eq:tobath} and \eqref{eq:result} say that
$ff\,(\GR{}\dg-\GR)$ is exactly $\GR\,\Sless_{A}\,\GA$ with the clean reservoir
self--energy \eqref{eq:result}. Contrast with the self--energies that were tried in
the code:

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lccc}
\toprule
$\Sless$ used & spectral in $\zeta^2\tz(\cdots)\tz$ & lead dependence & flat $2i\Gamma$?\\
\midrule
\textbf{FDT / $\Sless_{A}$ (correct)} & bare $\gz(i\Gamma)$ & none & \textbf{absent}\\
original (buggy) & \emph{dressed} $g_{r\alpha}-g_{a\alpha}$ & lead--dependent ($\Vimp^{L}\!\neq\!\Vimp^{R}$) & present\\
``clean $+2i\Gamma$'' & bare $\gz(i\Gamma)$ & none & present\\
``scheme B'' & bare $\gz(i0^{+})$ & none & present\\
\bottomrule
\end{tabular}
\end{center}

\noindent The correct $\Sless_{A}$ therefore differs from the original code in two ways,
both forced by the algebra above:
\begin{itemize}
\item \textbf{no $\Vimp$}: the impurity cancels in $\GR{}^{-1}-\GA{}^{-1}$, so it must
      not enter $\Sless$. Using the impurity--\emph{dressed} spectral made $\Sless$
      lead--dependent ($\Vimp^{L}\!\neq\!\Vimp^{R}$), which broke the antisymmetry
      $I(-\phi)=-I(\phi)$ and produced a spurious static diode ($\eta\sim-2.4\%$) and a
      $\sim\!1/\Gamma^{2}$--divergent, $\sim\!16\times$ inflated YSR supercurrent;
\item \textbf{no $2i\Gamma$}: by \eqref{eq:surfid} the whole broadening is already the
      reservoir term, so a separate flat $2i\Gamma$ has no retarded partner --- it is a
      double count.
\end{itemize}

\paragraph{Remark (scheme A vs.\ B).}
Keeping $\Gamma$ \emph{inside} $\gz(\omega+i\Gamma)$ (``scheme A'', the boxed result)
gives $\eta=0$ \emph{and} reproduces Ambegaokar--Baratoff for a clean junction,
$I_{c}eR_{N}/\Delta\to\pi/2\approx1.57$. Moving $\Gamma$ out of $\gz$ (evaluating the
reservoir at $i0^{+}$) and re--adding it as a separate intrinsic broadening $-i\Gamma$
on the contact --- with the matching retarded term
$\GR{}^{-1}=\gz(i0^{+})^{-1}+i\Gamma-\Vimp$ --- is also self--consistent ($\eta=0$) but
over--broadens the gap edge and \emph{fails} Ambegaokar--Baratoff
($I_{c}eR_{N}/\Delta\approx0.60$); it is therefore not used.

\paragraph{Bottom line.}
\[
  \texttt{Gljwar = ff .* ( Grjwar' - Grjwar )}
  \;\equiv\;
  \texttt{Grjwar * Siglj * Grjwar'}
  \ \ \text{with}\ \
  \Sless_{A}=ff\,\zeta^{2}\,\tz\bigl[-(\gz-\gz\dg)\bigr]\tz,\ \ \gz=\gz(\omega+i\Gamma).
\]
The FDT form simply computes this clean, impurity--free, $2i\Gamma$--free reservoir
self--energy without building it by hand.
